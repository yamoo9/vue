# 필터(Filters)

Vue.js는 텍스트 포멧을 변환하는 필터를 등록하고 사용할 수 있는 기능을 제공합니다.

예를 들어 숫자 텍스트 값을 한국 원화로 변경한다거나, 특정 접두사를 속성 앞에 붙이도록 설정할 경우 필터를 활용할 수 있습니다. 사용 방법은 아래와 같이 JavaScript 식 뒤에 파이프(|) 기호를 붙인 다음 필터를 호출합니다.

<!-- 콧수염(인터폴레이션) 사용 예 => '30,000원' -->
<span class="currency">{{ 30000 | currencyKor }}</span>

<!-- v-bind 디렉트브 사용 예 => 'yamoo9-note' -->
<div :id="'note' | assignPrefix">...</div>

# 글로벌 필터 등록

모든 컴포넌트에서 필터를 사용하고자 한다면 글로벌 방식으로 필터를 등록합니다.

main.js



 

import Vue from 'vue';

Vue.filter('ellipsis', value => `${value.slice(0,300)}...`);

# 로컬 필터 등록

특정 컴포넌트 내부에서만 필터를 사용할 경우 로컬 방식으로 필터를 등록합니다.

export default {
  filters: {
    currencyKor(value) {
      value = String(value)
                .split('')
                .reverse()
                .map((c,i) => i > 0 && i % 3 === 0 ? `${c},` : c)
                .reverse()
                .join('');
      return `${value}원`;
    }
  }
}

# 체이닝

필터는 여러 개의 필터를 연결하여 사용할 수 있습니다. 연결된 필터는 먼저 처리된 값을 첫번째 인자로 전달 받습니다.

<p>{{ 'sequential filters process the results chaining.' | capitalize | ellipsis }}</p>
1
filters: {
  capitalize(value) {
    return String(value)
            .split(' ')
            .map(word => word[0].toUpperCase() + word.slice(1))
            .join(' ');
  },
  ellipsis(value) {
    return String(value).slice(0, 15) + '...';
  }
}

# 전달인자

필터는 JavaScript 함수이므로 인자를 다수 전달할 수 있습니다. 필터 사용 시 설정한 인자는 2번째부터 전달됩니다.

<!-- 'y9-note'로 변환 -->
<div :id="'note' | assignPrefix('y9')">...</div>

<!-- 'It is abbr...'로 변환 -->
<p>{{ 'It is abbreviated by the value passed.' | ellipsis(10) }}</p>

# 다수 필터 등록

필터를 개별적으로 등록하는 방법도 있겠지만, 일괄적으로 자동 등록하도록 설정할 경우 아래와 같은 패턴을 사용합니다.

filters.js

export const ellipsis = (value, limit=300) => String(value).slice(0, limit) + '...';
export const assignPrefix(value, prefix='yamoo9') => `${String(prefix)}-${String(value)}`;
