# Progress Bar

Vue.js 애플리케이션에서 발생하는 페이지 트랜지션 또는 비동기 통신 과정의 진행 상태(progress)를 애니메이션 하는 Nprogress 라이브러리를 설치, 적용해봅니다.

# 설치

nprogress 패키지를 프로젝트에 설치합니다.

$ yarn add nprogress
1

# 스타일

Nprogress 스타일 모듈을 main.js에서 호출합니다.

// main.js
import 'nprogress/nprogress.css'
1
2

# 인터셉트

Axios를 사용해 구성한 API 서비스 파일(예: api/index.js)을 열어 Axios 인터셉트(Interceptors)를 사용해 비동기 통신의 시작(요청) 시점에 Nprogress를 시작(start)하고, 종료(응답) 시점에 완료(done) 하도록 설정합니다.





 













 




 



// api/index.js

import axios from 'axios'
// Nprogress 모듈 호출
import Nprogress from 'nprogress'

// Axios 인스턴스 생성
const apiService = axios.create({
  baseURL: '/api',
  withCredentials: false,
  headers: {
    Accept: 'application/json',
    'Content-Type': 'application/json'
  }
})

// Axios 인터셉트(요청)
apiService.interceptors.request.use((config) => {
  Nprogress.start() // 프로그레스 바 애니메이션 시작!
  return config
})
// Axios 인터셉트(응답)
api.interceptors.response.use((response) => {
  Nprogress.done() // 프로그레스 바 애니메이션 완료!
  return response
})
