# Axois 라이브러리

Axios는 Promise API를 활용하는 HTTP 비동기 통신 라이브러리입니다.

# 설치

axios 패키지를 프로젝트에 설치합니다.

 


$ yarn add axios
# npm i axios
1
2

# 사용법

컴포넌트에 axios 모듈을 불러온 후, 라이프 사이클 훅 함수, 메서드 등에서 사용할 수 있습니다.





 
 
 
 
 
 
 
 
 
 
 



import axios from 'axios';

export default {
  created() {
    axios
      // API 요청: GET, POST, PUT, DELETE
      .get('/api/members')
      // 응답(성공)
      .then(response =>
        console.log(response.data)
      )
      // 응답(실패)
      .catch(error =>
        console.log('오류 발생!!:', error.message)
      );
  }
}

json-server를 사용해 API를 요청 실습합니다.

# json-server 글로벌 설치

json-server 패키지를 글로벌 설치합니다.

 


$ yarn global add json-server
# npm i -g json-server

# 데이터베이스

데이터베이스로 사용할 db.json 파일을 준비한 후, json-server를 실행해 db.json을 지속적으로 관찰하도록 명령합니다.

$ json-server --watch db.json

# API 요청 및 처리

API를 요청해 응답 받은 데이터를 컴포넌트 데이터로 할당한 후, 컴포넌트 데이터를 순환하여 화면에 렌더링 합니다.










 











 
 
 
 




<template>
  <ul>
    <li v-for="item in lists" :key="item.id">
      {{ item.figure.name }}
    </li>
  </ul>
</template>

<script>
import axios from 'axios';

export default {
  name: 'FetchDataDemo',
  data: () => ({
    list: []
  }),
  created() {
    this.fetchData();
  },
  methods: {
    fetchData() {
      axios.get('http://localhost:3000/ediya-menu')
        .then(response => this.list = response.data)
        .catch(error => console.error('오류 발생!!', error.message));
    }
  }
}
</script>

# API 서비스

각 컴포넌트 마다 axios 모듈을 호출해 사용하는 것은 비 효율적입니다. 컴포넌트가 일괄적으로 접근할 수 있는 API 서비스를 제작해 사용하는 것이 효율적입니다.

apiService.js




 
 
 
 
 
 
 
 
 
 
 



 

 


import axios from 'axios';

// 싱글 Axios 인스턴스 생성
const api = axios.create({
  // Axios 인스턴스가 사용할 기본 URL 설정
  baseURL: 'http://localhost:3000/',
  // 인증(Authentication) 설정
  withCredentials: false,
  // 구성(Configuration) 설정
  headers: {
    'Accept': 'application/json',
    'Content-Type': 'application/json'
  }
});

export default {
  // 리스트 요청 메서드
  getEdiyaMenu: () => api.get('/ediya-menu'),
  // 개별 아이템 요청 메서드
  getEdiyaMenuItem: id => api.get(`/ediya-menu/${id}`),
};

각 컴포넌트는 단 하나의 서비스 객체를 통해 API를 요청할 수 있습니다.

전체 리스트를 요청하는 경우

MenuList.vue








 













 







<template>
  <ul class="menu-list">
    <menu-item v-for="item in list" :key="item.id" :id="item.id" />
  </ul>
</template>

<script>
import apiService from '@/apiService';
import MenuItem from '@/components/MenuItem';

export default {
  name: 'MenuList',
  components: { MenuItem },
  data: () => ({
    list: []
  }),
  created() {
    this.fetchData();
  },
  methods: {
    fetchData() {
      apiService.getEdiyaMenu()
        .then(response => this.list = response.data)
        .catch(error => console.error('오류 발생!!', error.message));
    }
  }
}
</script>

개별 아이템을 요청하는 경우

MenuItem.vue








 














 







<template>
  <li>
    {{ item.figure.name }}
  </li>
</template>

<script>
import apiService from '@/apiService';

export default {
  name: 'MenuItem',
  props: {
    id: {
      type: String,
      required: true
    }
  },
  created() {
    this.fetchData();
  },
  methods: {
    fetchData() {
      apiService.getEdiyaMenuItem(this.id)
        .then(response => this.item = response.data)
        .catch(error => console.error('오류 발생!!', error.message));
    }
  }
}
</script>
