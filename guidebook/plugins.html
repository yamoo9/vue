# 플러그인(Plugins)

Vue 플러그인은 Vue 애플리케이션을 확장하는 유용한 방법으로 빈번히 사용되는 기능을 묶은 패키지 입니다. 플러그인으로 분리된 기능은 여러 프로젝트에서 재사용할 수 있어 생산성과 사용성을 높입니다. 대표적인 플러그인은 Vue Router, Vuex 입니다.

# 플러그인 설치, 사용 설정

플러그인을 사용하려면 먼저 사용할 플러그인을 설치한 후, Vue.use() 메서드를 사용해 등록해야 합니다. 예를 들어 Vue Router 플러그인을 사용하고자 한다면 플러그인을 먼저 설치합니다.

 


$ yarn add vue-router
# 또는 npm i vue-router
1
2

설치한 후에는 설치한 플러그인을 불러와 Vue.use() 메서드를 사용해 등록 합니다. 플러그인이 설치 되면 Vue 인스턴스 내부에 플러그인 기능이 추가됩니다. 참고로 동일한 플러그인이 여러 번 등록되어도 플러그인은 한 번만 등록됩니다.

main.js




 





import Router from 'vue-router';

// `Router.install(Vue)` 실행
Vue.use(Router);

new Vue({
  // ...
});

# 플러그인 제작

플러그인을 가져다 사용하는 것을 넘어, 직접 제작해 사용하고 싶을 수도 있습니다. Vue 플러그인 제작 방법을 살펴봅시다.

MyPlugin.js

// 플러그인 객체
export default {

  // 플러그인 환경설정 객체
  config: { ... },

  // 플러그인 설치 함수
  install(Vue, options) {

    // Vue 클래스 메서드 추가
    Vue.classMethod() {
      // ...
    }

    // 글로벌 컴포넌트 추가
    Vue.component('컴포넌트-이름', {
      // ...
    });

    // 전역 디렉티브 추가
    Vue.directive('디렉티브-이름', {
      bind(el, binding, vnode, oldVnode) {
        // ...
      }
    });

    // 글로벌 믹스인 추가
    // ※ 모든 컴포넌트에 추가되니 주의!
    Vue.mixin({
      mounted(){
        // ...
      }
    });

    // 인스턴스 속성/메서드 추가
    Object.defineProperty(Vue.prototype, '속성-이름' {
      get () {
        // ...
      }
    });

    Object.defineProperty(Vue.prototype, '메서드-이름' {
      value () {
        // ...
      }
    });

  }

};

# GSAP 플러그인 예시

GSAP 라이브러리를 Vue 애플리케이션에서 사용할 수 있도록 등록하는 간단한 플러그인을 제작해보겠습니다. 먼저 Vue 프로토타입 객체의 $gsap 속성을 추가한 후, GSAP 라이브러리를 참조합니다.

// vue-gsap.js

import gsap from 'gsap';

export default {
  install(Vue) {
    Vue.prototype.$gsap = gsap;
  }
};

이제 플러그인을 프로젝트에 사용해 GSAP 애니메이션을 적용할 수 있습니다.














 














<template>
  <div id="app">
    <img ref="img" width="25%" src="./assets/logo.png">
  </div>
</template>

<script>
export default {
  name: "App",

  mounted() {

    const { img } = this.$refs;
    const { TimelineLite, Back } = this.$gsap;
    const timeline = new TimelineLite();

    timeline.to(img, 1.5, {
      x: 150,
      rotation: -180,
      ease: Back.easeInOut,
      delay: 1
    });

  },

};
</script>

# Vue Router 플러그인 예시

보다 실용적인 플러그인 제작 코드는 Vue Router 플러그인의 install.js를 살펴보세요.

// 플러그인 컴포넌트 호출
import View from './components/view'
import Link from './components/link'

// _Vue 내보내기
export let _Vue

// 플러그인 설치 함수
export function install (Vue) {

  // 플러그인을 이미 설치한 경우 함수 종료
  if (install.installed && _Vue === Vue) return

  // 플러그인 초기 설치 시, installed 값을 true로 변경
  install.installed = true

  // 플러그인 설치 시, 전달 받은 Vue 클래스 설정
  _Vue = Vue

  // 정의(Definition) 유무를 검사하는 헬퍼 함수
  const isDef = v => v !== undefined

  // 인스턴스 등록 함수
  const registerInstance = (vm, callVal) => {
    let i = vm.$options._parentVnode
    if (isDef(i) && isDef(i = i.data) && isDef(i = i.registerRouteInstance)) {
      i(vm, callVal)
    }
  }

  // 글로벌 믹스인 등록
  Vue.mixin({
    // 라이프 사이클 훅
    beforeCreate () {
      if (isDef(this.$options.router)) {
        this._routerRoot = this
        this._router = this.$options.router
        this._router.init(this)
        Vue.util.defineReactive(this, '_route', this._router.history.current)
      } else {
        this._routerRoot = (this.$parent && this.$parent._routerRoot) || this
      }
      registerInstance(this, this)
    },
    destroyed () {
      registerInstance(this)
    }
  })

  // Vue 인스턴스 속성 $router 확장
  Object.defineProperty(Vue.prototype, '$router', {
    get () { return this._routerRoot._router }
  })

  // Vue 인스턴스 속성 $route 확장
  Object.defineProperty(Vue.prototype, '$route', {
    get () { return this._routerRoot._route }
  })

  // 글로벌 컴포넌트 등록
  Vue.component('RouterView', View)
  Vue.component('RouterLink', Link)

  // 옵션 병합 전략 설정
  const strats = Vue.config.optionMergeStrategies
  // 라우트 훅(route hook)에 대해 동일한 훅 병합 전략 사용
  strats.beforeRouteEnter =
  strats.beforeRouteLeave =
  strats.beforeRouteUpdate = strats.created
}
