# 믹스인(Mixins)

컴포넌트를 개발하다 보면 컴포넌트 간에 중복되는 데이터, 라이프 사이클 훅, 메서드, 계산된 속성이 있을 수 있습니다. 중복되는 코드는 개발 관점에서 비효율적이므로 효율적인 관리가 필요해지게 됩니다. 믹스인(mixin)은 이런 경우 유용하게 사용됩니다.

믹스인은 말 그대로 "섞어 넣다"는 의미로 해석될 수 있습니다. 쉽게 말해 컴포넌트가 가진 공통 속성을 외부로 분리해 믹스인으로 관리하고, 믹스인을 컴포넌트에 등록하면 컴포넌트는 믹스인의 속성을 사용할 수 있게됩니다.


# 믹스인 정의

믹스인은 일반적인 JavaScript 객체로 Vue 컴포넌트가 가진 속성을 추가 할 수 있습니다.

// WhatIsMixin.js

export default {
  created() {
    this.logMessage();
  },
  data() {
    mixin_message: '믹스인은 중복되는 컴포넌트 속성을 분리 관리하여 재사용할 수 있도록 만들어 줍니다.'
  },
  methods: {
    logMessage() {
      console.log(this.mixin_message);
    }
  }
};

# 로컬 믹스인 등록

믹스인 등록 방법은 컴포넌트 등록 방법과 매우 유사합니다. 컴포넌트 mixins 속성에 사용할 믹스인을 불러와 등록해 사용합니다.


 



 



<script>
import WhatIsMixin from '@/mixins/WhatIsMixin.js';

export default {
  name: 'UsingMixinInComponent',
  mixins: [ WhatIsMixin ]
}
</script>

# 글로벌 믹스인 등록

믹스인 또한 컴포넌트와 마찬가지로 로컬, 글로벌 등록이 가능합니다. 글로벌 믹스인을 등록할 경우 모든 컴포넌트에 혼합됩니다. 예를 들어 사용자 정의 옵션(_$options)을 모든 컴포넌트에서 접근 사용할 경우 아래와 같이 작성할 수 있습니다.

// main.js

// Vue 루트 컴포넌트 생성 전에 글로벌 믹스인 등록
Vue.mixin({
  created() {
    // 모든 컴포넌트에 _$options 속성 할당
    this._$options = this.$options._$options;
  }
});

new Vue({
  // 모든 컴포넌트가 접근해 사용할 사용자 정의 옵션 객체 정의
  _$options: {
    animationOptions: { ... },
  },
  // ...
});

# 옵션 병합

등록된 믹스인과 컴포넌트가 중복된 라이프 사이클 훅, 데이터, 메서드를 가질 경우 적절한 옵션 병합 전략을 사용해 혼합됩니다.

  • 라이프 사이클 훅은 믹스인, 컴포넌트 모두 실행
  • 데이터 및 메서드는 컴포넌트 코드 실행 (믹스인 코드는 덮어씌어 짐)

# 라이프 사이클 훅 옵션 병합

믹스인 라이프 사이클 훅 함수와 컴포넌트 라이프 사이클 훅 함수가 중복될 경우 모두 실행됩니다.




 
 
 


// LifeCycleHookMixin.js

export default {
  mounted() {
    console.log('믹스인 mounted 훅 함수에서 호출');
  }
};

# 데이터, 메서드 옵션 병합

믹스인 데이터, 메서드와 컴포넌트가 중복될 경우 컴포넌트에 등록된 데이터와 메서드가 덮어씁니다.






 



 
 
 



// DataMixin.js

export default {
  data() {
    return  {
      isMixinRegistered: false
    }
  },
  methods: {
    logIsMixinRegistered() {
      console.log(`믹스인에서 실행: isMixinRegistered 값은 ${this.isMixinRegistered} 입니다.`);
    }
  }
};

# 활용 예시

폼 컨트롤 컴포넌트(base-input, base-select 등)에서 공통 사용할 코드를 믹스인에서 관리합니다.

// FormControlMixin.js

export default {
  inheritAttrs: false,
  props: {
    label: {
      type: String,
      default: ''
    },
    value: [String, Number]
  },
  methods: {
    updateValue(event) {
      this.$emit('input', event.target.value);
    }
  }
}
