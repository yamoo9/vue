# 비동기 컴포넌트

애플리케이션 규모가 커질 경우, 사용되지 않는 컴포넌트까지 모두 로딩하는 것은 성능에 부담을 줍니다. 이런 문제 해결을 위해 모든 컴포넌트를 로드하지 않고, 필요한 경우에만 특정 컴포넌트를 비동기 로드하는 방식을 사용해 성능을 향상시킬 수 있습니다.


# 로컬 컴포넌트에 등록

Vue 애플리케이션에서 필요한 상황에 비동기로 컴포넌트를 로딩하는 방법은 Webpack + ECMAScript 2015+ 구문을 사용합니다.








 



// 동기 로딩 컴포넌트
import syncComponent from '@/components/syncComponent';

export default {
  components: {
    syncComponent,
    // 비동기 로딩 컴포넌트
    'async-component': () => import('@/components/asyncComponent')
  }
}

# 글로벌 컴포넌트로 등록

비동기 컴포넌트를 글로벌로 등록하는 방법도 로컬 컴포넌트 등록과 유사합니다.



 

import Vue from 'vue';

Vue.component('async-component', () => import('@/components/asyncComponent'));

# 로딩 / 에러 설정

비동기 로딩 중, 로딩 스피너를 표시할 컴포넌트 또는 로드 실패 시 화면에 표시할 컴포넌트를 설정할 수 있습니다.





 

 













// 비동기 컴포넌트 팩토리 함수
const asyncComponent = () => ({
  component: import('@/components/asyncComponent'),
  // 로딩 스피너 컴포넌트 설정
  loading: LoadingSpinner,
  // 로딩에 실패하면 표시할 컴포넌트 설정
  error: DisplayErrorMessage,
  // 지연 시간 설정
  delay: 300,
  // 타임아웃 설정
  timeout: 400
});

export default {
  components: {
    // 비동기 컴포넌트 등록
    asyncComponent
  }
}
