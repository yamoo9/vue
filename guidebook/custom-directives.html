# 커스텀 디렉티브(Custom Directives)

Vue.js에서 기본 제공하는 디렉티브 뿐만 아니라, 사용자가 직접 정의한 디렉티브를 만들어 Vue 애플리케이션에 사용할 수 있습니다.


# 디렉티브 정의

사용자가 정의한 디렉티브 또한 v- 접두사가 붙습니다. 색상을 변경하는 간단한 v-color 디렉티브를 정의해봅니다.


 




<a
  v-color
  href="https://yamoo9.github.io/vue/guidebook/custom-directives.html">
  Vue 커스텀 디렉티브
</a>

Vue.js는 Vue.directive() 메서드를 사용해 사용자가 디렉티브를 정의할 수 있는 방법을 제공합니다. SwitchingColors.js 파일을 만들어 color 이름으로 요소의 색상을 변경하는 디렉티브를 정의합니다.





 
 
 

// SwitchingColors.js

import Vue from 'vue';

Vue.directive('color', (el) => {
  el.style.color = '#2c9dee';
});

로컬 디렉티브 정의

특정 컴포넌트 내부에서만 디렉티브를 사용 하도록 하려면 directives 속성을 컴포넌트에 추가해 사용합니다.









 
 
 



<script>
// 로컬 디렉티브
const colorDirective = (el) => {
  el.style.color = '#2c9dee';
};

export default {
  // 로컬 디렉티브 등록
  directives: {
    'color': colorDirective
  }
}
</script>

주의!

커스텀 디렉티브에 설정된 값은 JavaScript 식(expression)으로 유효한 데이터 타입을 전달해야 합니다.

커스텀 디렉티브에 전달된 값은 binding 매개변수 객체의 value 값으로 받습니다. 전달 받은 값을 요소에 설정하도록 코드를 작성합니다. 기존의 값은 기본(default) 값으로 설정합니다.




 


// SwitchingColors.js

Vue.directive('color', (el, binding) => {
  el.style.color = binding.value || '#2c9dee';
});

# 반복 변경되는 컬러 텍스트

반복해서 변경되는 컬러 텍스트 디렉티브로 변경해봅니다. 변경 처리할 컬러 리스트(배열)을 값으로 전달 설정합니다.

디렉티브 이름을 v-color에서 v-switching-colors로 변경한 후, 전달 받은 컬러 리스트를 순차적으로 요소의 색상으로 변경하도록 코드를 수정 작성합니다.

Vue.directive('switching-colors', (el, binding) => {
  const colors = binding.value;
  let i = 0;
  window.setInterval(() => {
    el.style.color = colors[i];
    if ( i > colors.length - 1 ) { i = 0 }
  }, 500);
});

# 반응성

Vue.js의 반응성은 디렉티브에서도 잘 작동합니다. 디렉티브가 설정된 이후에 데이터가 변경되면 변경된 데이터가 디렉티브에 잘 반영됩니다. 디렉티브의 반응성에 대해 살펴보도록 하죠. 디렉티브 값으로 설정했던 데이터를 컴포넌트의 데이터 속성 colors로 설정한 다음, 라이프 사이클 훅 함수를 사용해 컴포넌트 생성된 시점에 컬러 리스트에 아이템을 추가해봅니다.

배열 데이터와 반응성

Vue.js의 반응성을 유지하기 위해서는 배열 데이터의 아이템을 추가, 제거할 때 Array 객체의 메서드(push, splice)를 사용해야 합니다. 자세한 내용은 업데이트 관리 문서를 참고하세요.


디렉티브 코드는 마우스 인터랙션에 따라 작동하도록 colors 데이터의 아이템이 있을 경우와 그렇지 않은 경우로 나눠 처리하도록 작성합니다. 아이템이 없을 경우 더 이상 애니메이션이 되지 않도록 setInterval을 제거하고, 아이템이 있을 경우 setInterval을 설정합니다.

Vue.directive('switching-colors', (el, {value}) => {
  let i = 0;
  let colors = value || [];

  // interval을 제거할 ID
  let stopId;

  // colos 아이템이 있을 경우
  if (colors.length > 0) {
    stopId = window.setInterval(() => {
      el.style.color = colors[i++ % colors.length];
      if (i > colors.length - 1) { i = 0; }
    }, 600);
  }
  // colos 아이템이 없을 경우
  else {
    // 초기화
    i = 0;
    window.clearInterval(stopId);
  }

});

# 전달인자

디렉티브는 값 뿐만 아니라, 인자(argument)를 전달 받을 수도 있습니다. 속도(speed) 키워드를 전달해 애니메이션 속도를 제어해 봅시다. 설정할 키워드는 다음과 같습니다.

키워드 값(ms)
slow 1000
normal 600
fast 300
crazy 100

디렉티브 전달인자는 콜론(:) 뒤에 추가합니다. (v-switching-colors:{키워드})


 






<a
  v-switching-colors:crazy="colors"
  @mouseenter="setColors"
  @mouseleave="resetColors"
  href="#">
  Crazy
</a>

디렉티브 코드에 스피드 키워드 리스트 speeds 객체를 추가한 다음 전달인자를 받아 처리할 수 있도록 설정합니다.


 
 
 
 
 
 



 







 







 









// 스피드 키워드 리스트
const speeds = {
  slow: 800,
  normal: 500,
  fast: 300,
  crazy: 100
};

Vue.directive('switching-colors',
  // 구조 분해 할당을 사용해 binding 객체의 arg 속성 추가
  (el, {value, arg}) => {

    let i = 0;
    let colors = value || [];
    let stopId;

    // arg 값이 전달될 경우, 스피드 키워드 리스트에서 값을 도출
    // arg 값이 없을 경우, 기본 값 사용
    let speed = speeds[ arg || 'normal' ];

    if (colors.length > 0) {
      stopId = window.setInterval(() => {
        el.style.color = colors[i++ % colors.length];
        if (i > colors.length - 1) { i = 0; }
      },
      // 스피드 설정
      speed);
    }
    else {
      i = 0;
      window.clearInterval(stopId);
    }

  }
);

디렉티브 옵션

전달인자와 수정자는 옵션이므로 생략해도 무방합니다.

디렉티브 코드는 전달된 수정자를 감지하여 스위칭 처리하도록 다음과 같이 작성합니다.



 


 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 



Vue.directive('switching-colors',
  // 수정자(modifiers) 속성을 구조 분해 할당 구문 안에 추가
  (el, { value, arg, modifiers }) => {
  // ...
  // modifiers 객체를 순환 처리하는 for-in 문
  for ( let key in modifiers ) {
    // 자신의 속성만 순환 처리
    if ( modifiers.hasOwnProperty(key) ) {
      // key 값에 따라 스위칭
      switch(key) {
        // 수정자가 box인 경우
        case 'box':
          el.style.cssText += `
            padding: 0.3em 0.54em;
            border: 1px solid currentColor;
          `;
        break;
        // 수정자가 roundBox인 경우
        case 'roundBox':
          el.style.cssText += `
            padding: 0.3em 0.54em;
            border: 1px solid currentColor;
            border-radius: 6px;
          `;
        break; thick
        // 수정자가 thick인 경우
        case 'weight':
          el.style.cssText += `
            border-width: 4px;
          `;
          break;
        default:
          el.style.cssText += `padding: 0.3em 0.54em`;
      }
    }
  }
  // ...
}

# 훅 함수

컴포넌트가 라이프 사이클 훅을 가지고 있는 것처럼 디렉티브도 훅 함수를 가집니다. 디렉티브 훅 함수의 종류는 다음과 같습니다.

훅 함수 설명
bind 디렉티브가 처음 요소에 바인딩 될 때 1회 호출됩니다.
inserted 디렉티브가 바인딩 된 요소가 부모 노드에 삽입된 후 호출됩니다. (부모노드 존재 보장)
update 디렉티브를 포함하는 컴포넌트가 업데이트 된 후 호출됩니다.
(단, 자식 노드가 업데이트 되기 전에 호출)
componentUpdate 자식 노드 업데이트를 포함해서 컴포넌트가 업데이트 된 후 호출됩니다.
unbind 디렉티브가 요소에서 바인딩 해제될 때 1회 호출됩니다.

훅 함수를 추가하는 방법은 다음과 같습니다. 훅 함수는 옵션이므로 모두 추가할 필요는 없습니다.

Vue.directive('detection-hook', {
  bind(el, binding) {
    console.log('bind 훅');
  },
  inserted(el, binding) {
    console.log('inserted 훅');
  },
  update(el, binding) {
    console.log('update 훅');
  },
  componentUpdate(el, binding) {
    console.log('componentUpdate 훅');
  },
  unbind(el, binding) {
    console.log('unbind 훅');
  },
})

# setInterval 애니메이션 제거

작성한 v-switching-colors 디렉티브는 setInterval을 사용하고 있어 많이 사용할 경우 성능에 악 영향을 미칩니다. 특히 디렉티브가 설정된 요소가 제거(해제)된 경우에도 setInterval은 계속 돌아갈 경우 문제가 됩니다. 이런 경우를 방지하기 위해 unbind 훅 함수를 사용해 요소가 제거되면 clearInterval을 실행해 애니메이션을 중단해야 합니다.


unbind 훅 함수 사용 예를 보기 위해 코드를 다소 수정했습니다. 이벤트 인터랙션은 제거하고, 바인드 된 시점부터 자동 반복해서 텍스트 컬러가 변경됩니다. 토글 버튼을 추가해서 버튼 클릭 이벤트에 따라 디렉티브가 바인딩 된 요소들을 제거하거나, 추가합니다.








 


 
 
 
 
 
 
 
 
 









 






<template>
  <div id="app">

    <p>
      Console 패널을 열고 토글 버튼을 누르면 <br>
      setInterval 애니메이션이 정지 또는 재생되는 것을<br>
      확인할 수 있습니다.<br>
      <button class="toggle-button" @click="isVisible = !isVisible">토글</button>
    </p>

    <template v-if="isVisible">
      <a v-switching-colors:crazy.box="colors" href="#">
        Crazy
      </a>
      <hr>
      <a v-switching-colors.roundBox.thick="colors" href="#">
        Slow
      </a>
    </template>

  </div>
</template>

setInterval 코드 부분을 아래와 같이 수정하고, 인터벌 ID 값을 el 객체의 속성 __stopId로 설정합니다. unbind 훅 함수를 추가한 다음 clearInterval을 사용해 el.__stopId__로 식별되는 인터벌 애니메이션을 제거합니다.

const bindUpdateHookFn = (el, { value, arg, modifiers }) => {
  let i = 0;
  let colors = value || [];
  let speed = speeds[arg || 'normal'];

  for (let key in modifiers) {

  }

  el.__stopId__ = window.setInterval(() => {
    console.log('🎨 컬러링 중...');
    el.style.color = colors[i++ % colors.length];
    if (i > colors.length - 1) { i = 0; }
  }, speed);

Vue.directive('switching-colors', {
  bind: bindUpdateHookFn,
  update: bindUpdateHookFn,
  unbind(el) {
    window.clearInterval(el.__stopId__);
  }
});

# click-outside 디렉티브

특정 영역을 클릭 후, 영역 밖으로 끌고 나가 놓았을 때 감지하는 이벤트를 처리하는 디렉티브를 만들어 봅니다.

<template>
  <div class="box" v-click-outside="onClickOutside">
    클릭 후, 박스 영역 밖으로 끌고 나가 높으면 이벤트가 발생합니다.
  </div>
</template>

<script>
export default {
  methods: {
    onClickOutside() {
      window.alert('영역 밖으로 이동한 후 놓았습니다. 🚀');
    }
  }
};
</script>

Vue.directive('click-outside', {
  bind(el, { value }) {
    // unbind 훅에서 활용할 수 있도록 설정
    el.__clickOutsideHandler__ = e => {
      // 클릭한 요소 또는 내부에 포함하는 요소가 아닐 경우, value() 함수 실행
      if ( !(el === e.target || el.contains(e.target)) ) {
        value();
      }
    };
    // 이벤트 추가
    document.body.addEventListener('click', el.__clickOutsideHandler__);
  },
  unbind(el) {
    // 이벤트 제거
    document.body.removeEventListener('click', el.__clickOutsideHandler__);
  }
});
