# 단방향 데이터 흐름

Vue 애플리케이션은 기본적으로 단방향(1-way)으로 데이터를 전달하는 흐름을 가지므로, 전달받은 부모 컴포넌트의 속성을 임의로 변경 시도할 경우 다음과 같은 오류가 발생 합니다.

오류!

[Vue warn]: Avoid mutating a prop directly since the value will be overwritten whenever the parent component re-renders. Instead, use a data or computed property based on the prop's value. Prop being mutated: "me"

# 속성 복사

이런 경우 전달 받은 부모 컴포넌트의 속성을 로컬 데이터 또는 계산된 속성을 사용해 자식 컴포넌트로 복사한 다음 임의로 조작/변경 할 수 있습니다. 변경된 속성은 자식 컴포넌트의 속성으로 부모 컴포넌트의 속성은 변경되지 않습니다.








 





 




export default {
  props: {
    me: Object
  },
  // 방법 1. 로컬 데이터 설정(객체 복사)
  data() {
    return {
      iam: { ...this.me, isLocalData: true }
    }
  },
  // 방법 2. 계산된 속성 설정(객체 복사)
  computed: {
    itsMe() {
      return { ...this.me, isComputed: true };
    }
  }
}

# $parent 속성

자식 컴포넌트에서 부모 컴포넌트의 속성을 직접 변경하고 싶다면? 자식 컴포넌트의 $parent 속성과 Vue.set() 메서드를 사용해 부모 컴포넌트의 데이터 값을 직접 변경할 수 있습니다.







 









export default {
  props: {
    me: Object
  },
  methods: {
    changeParentProp() {
      this.$parent.me.changedMessage: '자식 컴포넌트가 부모 컴포넌트의 속성을 바꿉니다.';
      // 반응성 설정
      // Vue.set(this.$parent.me, {
      //   ...this.me,
      //   changedMessage: '자식 컴포넌트가 부모 컴포넌트의 속성을 바꿉니다.'
      // });
    }
  }
}

# 이벤트 방출

자식 컴포넌트가 부모 컴포넌트에 속성 변경을 위한 이벤트를 방출(emit) 해야 합니다. 부모 컴포넌트에 이벤트를 방출하려면 자식 컴포넌트의 인스턴스 메서드 .emit()을 사용합니다.













 















<script>
export default {
  data: () => ({
    input: { name: '', description: '' }
  }),
  props: {
    index: Number,
    me: Object
  },
  methods: {
    releaseEventToParent() {
      // 사용자 이벤트 changeMe를 부모 컴포넌트에 방출합니다.
      this.$emit('changeMe', this.index, { ...this.input });
    }
  }
}
</script>

<template>
  <div class="child-component">
    <form @submit.prevent="releaseEventToParent">
      <label>이름: <input v-model="input.name"></label>
      <label>설명: <text-area v-model="input.description"></text-area></label>
      <button>부모에게 이벤트 방출</button>
    </a>
  </div>
</template>

# 이벤트 감지

자식 컴포넌트가 방출한 커스텀 이벤트를 부모 컴포넌트는 감지한 후, 자신의 메서드를 실행해 부모 자신의 속성을 변경할 수 있습니다. 부모 컴포넌트가 자신의 속성을 변경하면, 반응성 시스템에 의해 자식 컴포넌트는 업데이트 됩니다.

부모 컴포넌트








 








 
 
 




<template>
  <div class="parent-component">
    <child-component
      v-for="(child,i) of children"
      :key="i"
      :index="i"
      :me="child"
      @changeMe="onChangeChild"
    />
  </div>
</template>

<script>
export default {
  // ...
  methods: {
    onChangeChild(index, newChild) {
      this.children.splice(index, 1, newChild);
    }
  }
}
</script>

# .sync 수식어

부모 컴포넌트가 자식 컴포넌트에 속성 전달할 때 .sync 수식어를 사용하면, 자식 컴포넌트에서 방출하는 이벤트를 감지하는 별도의 코드 없이 데이터를 변경할 수 있도록 설정할 수 있습니다. (양방향 바인딩)




 














<template>
  <div class="parent-component">
    <child-component
      :choco.sync="chocolet"
    />
  </div>
</template>

<script>
export default {
  data() {
    return {
      chocolet: '가나 초콜렛'
    }
  }
}
</script>

this.$emit('update:choco', '크런키 초콜렛');

# .native 수식어

컴포넌트에 직접 이벤트를 바인딩 할 경우, 이벤트가 작동하지 않습니다.



 













<template>
  <div class="native-modifier-demo">
    <component is="comp" @click="logEventBind" />
  </div>
</template>

<script>
export default {
  methods: {
    logEventBind() {
      console.log('컴포넌트 이벤트 바인딩');
    }
  }
}
</script>

# $listeners 속성

컴포넌트에 이벤트를 바인딩 할 때 .native 수식어를 사용하지 않고도, 컴포넌트에서 이벤트를 바인딩 할 방법이 있습니다. 컴포넌트 템플릿 루트 요소에 v-on 디렉티브를 설정하고, $listeners 값을 설정하면 됩니다.




 





<template>
  <div
    class="root-in-component"
    v-on="$listeners"
  >
    <!-- ... -->
  </div>
</template>

# 컴포넌트 + v-model

앞서 공부한 사용자 정의 이벤트는 v-model로 작동되는 컴포넌트를 만드는데 사용할 수 있습니다. 일반적인 HTML <input> 요소에 v-model 디렉티브를 설정하면 입력과 동시에 데이터에 반영되는 양방향 바인딩이 설정됩니다.

<input v-model="price">
1

위 v-model 코드와 아래 코드는 동일합니다.

<input
  :value="price"
  @input="price = $event.target.value"
>
1
2
3
4

이를 컴포넌트에 활용하면 다음과 같이 작성할 수 있습니다.

<currency-input
  :value="price"
  @input="value => { price = value }"
/>
1
2
3
4

위 코드는 부모 컴포넌트에서 자식 컴포넌트에 속성을 전달하고, 자식 컴포넌트가 이벤트를 방출해 부모가 이벤트를 감지해 처리하는 코드와 비슷합니다.

  • 전달 속성(props): price
  • 방출 이벤트(emit Event): input

이를 활용하면 컴포넌트에 양방향 바인딩 코드를 작성할 수 있습니다.

부모 컴포넌트

<currency-input v-model="price"/>
1

자식 컴포넌트

props: {
  // 부모 컴포넌트에서 value 속성 받기
  value: {
    type: String,
    default: 0
  }
},
methods: {
  changeValue(newValue) {
    // 부모 컴포넌트에 @input 이벤트 방출 (value 전달)
    this.$emit('input', newValue);
  }
}

# 컴포넌트 + 커스텀 v-model

기본적으로 컴포넌트의 v-model은 value를 전달 받고, input 이벤트를 방출하지만, 체크 박스와 라디오 버튼과 같은 일부 입력 컨트롤은 다른 목적으로 value 속성을 사용하고 싶을 수 있습니다. 이런 경우 v-model 대신 아래와 같이 전달 속성과 방출 이벤트를 임의로 설정해 양방향 바인딩 할 수 있습니다.

<app-checbox
  :checked="isChecked"
  @change="value => { isChecked = value }"
/>
1
2
3
4

하지만 v-model을 그대로 사용하면서 사용자가 임의로 전달 속성과 방출 이벤트 타입을 설정할 수 있다면 편리할 겁니다. 컴포넌트의 model 속성에 전달 속성, 방출 이벤트 타입을 설정하면 v-model을 사용자가 임의로 조정할 수 있습니다.

export default {
  model {
    // 전달 받는 속성
    prop: 'checked',
    // 방출할 이벤트 타입
    event:'change'
  },
  // 부모에게 전달 받을 속성
  props: {
    checked: Boolean, // v-model이 사용
    value: String
  }
}
